%% Lagarange - Curve dila nasl tr particular value sathi y kase find out 

x = [0 1 2 3 4];
y = [1 2 5 10 17];

X = 3.5;

n = length(x);  

P = 0;

for i = 1:n  
    L = 1;
    for j = 1:n
        if i ~= j
            L = L * ((X - x(j)) / (x(i) - x(j)));
        end
    end
    P = P + y(i) * L;
end

display(P)
