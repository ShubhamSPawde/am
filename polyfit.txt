%% Graph Poly fit
x = [2 4 6 8 10 12 14]
y = [-7 19 33 104 213 9 23]
hold on
plot(x, y, '+', LineWidth=5)
p = polyfit(x, y, 6)
x_fit = [1:0.1:15]
y_fit = polyval(p, x_fit)
plot(x_fit, y_fit)
hold off

%% 
x = [1 2 3 4 5 6 7]
y = [2 5 24 61 122 213 340]
hold on
ylog = log(y)
plot(x, y, '*', Color='r')
p = polyfit(x, ylog, 1)
b = p(1)
a = exp(p(2))
y_fit = a.*exp(b.*x)
plot(x, y_fit)
