%% Rk Method

syms f(x, y)
f(x, y) = x / y;

atwhichpointuwanttofind = 4; 

x = 1;
y = 2;
h = 0.1;
n = (atwhichpointuwanttofind - x(1)) / h;

for i=2 : (n+1)
    k1=0; k2=0; k3=0; k4=0;
    x(i) = x(i-1) + h;
    k1 = h*f(x(i-1), y(i-1));
    k2 = h*f(x(i-1) + h/2 , y(i-1) + k1/2);
    k3 = h*f(x(i-1) + h/2 , y(i-1) + k2/2);
    k4 = h*f(x(i-1) + h/2 , y(i-1) + k3/2);
    y(i) = y(i-1) + (1/6) * (k1 + 2*k2 + 2*k3 + k4);
end

display("Rk ans : " + y(n+1))
