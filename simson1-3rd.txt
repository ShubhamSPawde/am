%% Simpson''s 1/3 Rule

x = input('Enter x values: ');
y = input('Enter y values: ');
n = length(x);


h = (x(end) - x(1)) / (n-1);
s1 = 0;
s2 = 0;
for i = 2:2:n-1
   s1 = s1 + y(i);
end
for i = 3:2:n-2
   s2 = s2 + y(i);
end
result = (h/3) * (y(1) + y(end) + 4*s1 + 2*s2);
disp('Result using Simpson''s 1/3 Rule ='), disp(result);
